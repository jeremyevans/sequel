def [](v)
  case v
  when Symbol, Sequel::SQL::Identifier, Sequel::SQL::QualifiedIdentifier
    Sequel::SQL::QualifiedIdentifier.new(self, v)
  else
    aref_before_sequel(v)
  end
end
