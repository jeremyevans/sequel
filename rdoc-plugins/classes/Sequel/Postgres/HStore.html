unquoted_literal ()

Return a string containing the unquoted, unstring-escaped literal version of the hstore. Separated out for use by the bound argument code.

[show source] 
# File lib/sequel/extensions/pg_hstore.rb, line 269
def unquoted_literal
  str = String.new
  comma = false
  commas = COMMA
  quote = QUOTE
  kv_sep = KV_SEP
  null = NULL
  each do |k, v|
    str << commas if comma
    str << quote << escape_value(k) << quote
    str << kv_sep
    if v.nil?
      str << null
    else
      str << quote << escape_value(v) << quote
    end
    comma = true
  end
  str
end
