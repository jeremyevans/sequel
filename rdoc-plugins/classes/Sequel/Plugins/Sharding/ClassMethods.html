def eager_load_results(opts, eo, &block)
  if (s = eo[:self]) && (server = s.opts[:server])
    eb = eo[:eager_block]
    set_server = proc do |ds|
      ds = eb.call(ds) if eb
      ds = ds.server(server) unless ds.opts[:server]
      ds
    end
    eo = eo.merge(:eager_block=>set_server)
  end

  super
end
