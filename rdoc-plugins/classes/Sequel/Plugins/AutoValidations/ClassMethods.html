def skip_auto_validations(type)
  if type == :all
    [:not_null, :types, :unique, :max_length].each{|v| skip_auto_validations(v)}
  elsif type == :types
    @auto_validate_types = false
  else
    send("auto_validate_#{type}_columns").clear
  end
end
