def load_typecast_pg(values)
  pg_typecast_on_load_columns.each do |c|
    if (v = values[c]).is_a?(String) && (oid = db_schema[c][:oid]) && (pr = db.conversion_procs[oid])
      values[c] = pr.call(v)
    end
  end
  values
end
