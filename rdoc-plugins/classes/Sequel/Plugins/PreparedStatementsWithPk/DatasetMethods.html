def with_pk(pk)
  begin
    ds, bv = unbind
  rescue UnbindDuplicate
    super
  else
    begin
      bv = bv.merge!(model.primary_key_hash(pk)){|k, v1, v2| ((v1 == v2) ? v1 : raise(UnbindDuplicate))}
    rescue UnbindDuplicate
      super
    else
      model.send(:prepared_lookup_dataset, ds).call(bv)
    end
  end
end
