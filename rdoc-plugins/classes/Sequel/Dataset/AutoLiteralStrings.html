def filter_expr(expr = nil)
  case expr
  when LiteralString
    super
  when String
    super(LiteralString.new(expr))
  when Array
    if (sexpr = expr.first).is_a?(String)
      super(SQL::PlaceholderLiteralString.new(sexpr, expr[1..-1], true))
    else
      super
    end
  else
    super
  end
end

def update_sql(values=OPTS)
  case values
  when LiteralString
    super
  when String
    super(LiteralString.new(values))
  else
    super
  end
