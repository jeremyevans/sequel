check_disconnect_errors()

Raise a Sequel::DatabaseDisconnectError if a one of the disconnect error classes is raised, or a PGError is raised and the connection status cannot be determined or it is not OK.

[show source] 
    # File lib/sequel/adapters/postgres.rb
118 def check_disconnect_errors
119   yield
120 rescue *DISCONNECT_ERROR_CLASSES => e
121   disconnect = true
122   raise(Sequel.convert_exception_class(e, Sequel::DatabaseDisconnectError))
123 rescue PGError => e
124   disconnect = false
125   begin
126     s = status
127   rescue PGError
128     disconnect = true
129   end
130   status_ok = (s == Adapter::CONNECTION_OK)
131   disconnect ||= !status_ok
132   disconnect ||= e.message =~ DISCONNECT_ERROR_RE
133   disconnect ? raise(Sequel.convert_exception_class(e, Sequel::DatabaseDisconnectError)) : raise
134 ensure
135   block if status_ok && !disconnect
136 end
