def execute(sql, opts=OPTS)
  synchronize(opts[:server]) do |conn|
    begin
      res = log_connection_yield(sql, conn){conn.execute(sql)}
      yield res if block_given?
      nil
    rescue ::Swift::Error => e
      raise_error(e)
    end
  end
end
