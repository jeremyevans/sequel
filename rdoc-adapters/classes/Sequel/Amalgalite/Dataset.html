def fetch_rows(sql)
  execute(sql) do |stmt|
    @columns = cols = stmt.result_fields.map{|c| output_identifier(c)}
    col_count = cols.size
    stmt.each do |result|
      row = {}
      col_count.times{|i| row[cols[i]] = result[i]}
      yield row
    end
  end
end
