def select_limit_sql(sql)
  if l = @opts[:limit]
    if l == 1
      sql << " FETCH FIRST ROW ONLY"
    elsif l > 1
      sql << " FETCH FIRST "
      literal_append(sql, l)
      sql << " ROWS ONLY"
    end
  end
end
