def subselect_sql_append(sql, ds)
  ds.clone(:append_sql=>sql, :prepared_args=>prepared_args, :bind_vars=>@opts[:bind_vars]).
    send(:to_prepared_statement, :select, nil, :extend=>bound_variable_modules).
    prepared_sql
end
