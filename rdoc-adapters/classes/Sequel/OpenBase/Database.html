def connect(server)
  opts = server_opts(server)
  OpenBase.new(
    opts[:database],
    opts[:host] || 'localhost',
    opts[:user],
    opts[:password]
  )
end

def execute(sql, opts=OPTS)
  synchronize(opts[:server]) do |conn|
    r = log_yield(sql){conn.execute(sql)}
    yield(r) if block_given?
    r
  end
end
